void
NixVectorRouting::SetPaths(std::string pathFile)
{
    std::ifstream file(pathFile);
    if (!file.is_open())
    {
        NS_LOG_ERROR("Cannot open path file: " << pathFile);
        return;
    }
    
    std::string line;
    while (std::getline(file, line))
    {
        std::istringstream iss(line);
        int srcId, dstId, pathLength;
        
        // 读取 srcId, dstId, pathLength
        iss >> srcId >> dstId >> pathLength;
        
        // 读取完整路径
        std::vector<int> path;
        int nodeId;
        for (int i = 0; i < pathLength; i++)
        {
            iss >> nodeId;
            path.push_back(nodeId);
        }
        
        // 存储到 Table 中
        std::pair<int, int> key = std::make_pair(srcId, dstId);
        Table[key] = path;
    }
    
    file.close();
}
